65 std::vector<float> quadratic_equation(float a, float b, float c)
66 {
67  std::vector<float> result;
68  float discriminant = pow(b,2) - (4 * a * c);
69  if(discriminant < 0) // No result
70  {
71  return result;
72  }
73  else if(discriminant == 0) //1 result
74  {
75  result.push_back(-b/(2*a));
76  }
77  else
78  {
79  result.push_back(((-b + sqrt(discriminant))/ (2 * a)));
80  result.push_back(((-b - sqrt(discriminant))/ (2 * a)));
81  }
82  return result;
83 }
