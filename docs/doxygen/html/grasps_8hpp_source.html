142  void get_checkpoints(Msg msg)
143  {
144 
145 
146  for(int i = 0; i < msg.num_objects; i++) //Iterate through the detected objects
147  {
148  std::vector<int> temp_checkpoint_startx;
149  std::vector<int> temp_checkpoint_starty;
150  std::vector<int> temp_checkpoint_endx;
151  std::vector<int> temp_checkpoint_endy;
152 
153  //1 Deg is 0.0174533 rad
154  float gradient_perpendicular;
155  float gradient;
156  int search_area_start_x ;
157  int search_area_start_y ;
158  int search_area_end_x ;
159  int search_area_end_y ;
160  float c1,c2;
161  c1 = 0;
162  c2 = 0;
163 
164  //Determine search area for gripping start and end points. These search areas will only be used at the end of this function
165  if(abs(msg.br_y[i]- msg.tl_y[i]) < distance_between_fingers ) //If the gripper is bigger than the bounding box, the search area needs to be beyond the bounding box points
166  {
167  search_area_start_y = msg.tl_y[i] - int(round(abs(distance_between_fingers - abs(msg.br_y[i]- msg.tl_y[i]))/2)) ;
168  search_area_end_y = msg.br_y[i] + int(round(abs(distance_between_fingers - abs(msg.br_y[i]- msg.tl_y[i]))/2)) ;
169  }
170  else //If not the search box can just be the bounding box
171  {
172  search_area_start_y = msg.tl_y[i];
173  search_area_end_y = msg.br_y[i];
174  }
175  if(abs(msg.br_x[i]- msg.tl_x[i]) < distance_between_fingers ) //If the gripper is bigger than the bounding box, the search area needs to be beyond the bounding box points
176  {
177  search_area_start_x = msg.tl_x[i] - int(round(abs(distance_between_fingers - abs(msg.br_x[i]- msg.tl_x[i]))/2)) ;
178  search_area_end_x = msg.br_x[i] + int(round(abs(distance_between_fingers - abs(msg.br_x[i]- msg.tl_x[i]))/2)) ;
179  }
180  else //If not the search box can just be the bounding box
181  {
182  search_area_start_x = msg.tl_x[i];
183  search_area_end_x = msg.br_x[i];
184  }
185 
186  //Ensure that the coordinates are valid within the image viewing space
187  if(search_area_start_x < 0)
188  {
189  search_area_start_x = 0;
190  }
191  if(search_area_start_y < 0)
192  {
193  search_area_start_y = 0;
194  }
195  if(search_area_end_x > msg.img_width)
196  {
197  search_area_end_x = msg.img_width;
198  }
199  if(search_area_end_y > msg.img_height)
200  {
201  search_area_end_y = msg.img_height;
202  }
203 
204  //Get the gradient value of the object
205  gradient = -tan(msg.angle[i]); // NOTE:: negative here because the x-y axis of an image is flipped about the x axis
206 
207  //Get the gradient value of the perpendicular line from object
208  gradient_perpendicular = -(1/gradient);
209 
210  float c_perpendicular;
211  float r = distance_between_fingers/2;
212  int x1,x2,y1,y2;
213  //Get y-intercept of perpendicular line
214  c_perpendicular = msg.center_y[i] - (gradient_perpendicular) * msg.center_x[i];
215 
216  if(abs(gradient) < min_zero_angle) // When the obect is horizontal WRT x axis, angle 0
217  {
218  allowable_length_error = 1;
219  if(msg.center_y[i] -r < 0) //the object is too close to the edge of the viewing range of the camera
220  {
221  y1 = 0;
222  }
223  else
224  {
225  y1 = msg.center_y[i] -r;
226  }
227 
228  y2 =y1 + 2 * r;
229 
230  // For Horizontal objects, search area can be reduced to just adding all the x values along the horizontal lines
231  for(int j = search_area_start_x; j < search_area_end_x + 1; j++)
232  {
233  if(y2 < msg.img_height)
234  {
235  temp_checkpoint_endy.push_back(y2);
236  }
237  else
238  {
239  temp_checkpoint_endy.push_back(msg.img_height);
240  }
241  if(j < msg.img_width)
242  {
243  temp_checkpoint_endx.push_back(j);
244  temp_checkpoint_startx.push_back(j);
245  }
246  else
247  {
248  temp_checkpoint_endx.push_back(msg.img_width);
249  temp_checkpoint_startx.push_back(msg.img_width);
250  }
251 
252  if(j > 0)
253  {
254  temp_checkpoint_endx.push_back(j);
255  temp_checkpoint_startx.push_back(j);
256  }
257  else
258  {
259  temp_checkpoint_endx.push_back(0);
260  temp_checkpoint_startx.push_back(0);
261  }
262  //
263  if(y1 >= 0)
264  {
265  temp_checkpoint_starty.push_back(y1);
266  }
267  else
268  {
269  temp_checkpoint_starty.push_back(0);
270  }
271  }
272  }
273  else if(abs(gradient_perpendicular) < min_zero_angle) // object is perpendicular wrt x axis, angle pi/2
274  {
275  allowable_length_error = 1;
276  if(msg.center_x[i] - r < 0)
277  {
278  x2 = 0; //Get First x bound
279  }
280  else
281  {
282  x2 = msg.center_x[i] - r;
283  }
284  x1 = x2 + 2 * r; //Get Second x bound
285  // For Horizontal objects, search area can be reduced to just adding all the y values along the vertical lines
286  for(int j = search_area_start_y; j < search_area_end_y + 1; j++)
287  {
288 
289  if(j < msg.img_height)
290  {
291  temp_checkpoint_endy.push_back(j);
292  temp_checkpoint_starty.push_back(j);
293  }
294  else
295  {
296  temp_checkpoint_starty.push_back(msg.img_height);
297  temp_checkpoint_endy.push_back(msg.img_height);
298  }
299  if(j > 0)
300  {
301  temp_checkpoint_endy.push_back(j);
302  temp_checkpoint_starty.push_back(j);
303  }
304  else
305  {
306  temp_checkpoint_starty.push_back(0);
307  temp_checkpoint_endy.push_back(0);
308  }
309  if(x2 < msg.img_width)
310  {
311  temp_checkpoint_endx.push_back(x2);
312  }
313  else
314  {
315  temp_checkpoint_endx.push_back(msg.img_width);
316  }
317  if(x1 >= 0)
318  {
319  temp_checkpoint_startx.push_back(x1);
320  }
321  else
322  {
323  temp_checkpoint_startx.push_back(0);
324  }
325  }
326  }
327  else // If the object is angled
328  {
329  allowable_length_error = 3;
330  /*
331  Given a circle with equation (x- centre[0])^2 + (y-centre[1])^2 = r^2, where r = gripper_width/2
332  Find the two points where the line y = gradient_perpendicular* x + c intersect
333  */
334 
335  std::vector<std::vector<int>> intersect_temp = circle_line_intersect(gradient_perpendicular, c_perpendicular, r, {msg.center_x[i] , msg.center_y[i]});
336  /*
337  Problem with using the function to generate the intersect points is that it comes as an int, so the c1 values created are abit truncated,
338  compared to if the intersect points were kept as floats
339  */
340 
341  c1 = intersect_temp[0][1] - gradient * intersect_temp[0][0];
342  c2 = intersect_temp[1][1] - gradient * intersect_temp[1][0];
343 
344  /*
345  Now that we know the two lines that represents the max and min bounds of the gripper.
346  Get the points in the search area that are along these lines.
347  Assumes y1 = m1x + c1 to be the end corner of the grasps
348  Assumes y2 = m2x + c2 to be the end corner of the grasps
349  */
350 
351  for(int m = search_area_start_y; m < search_area_end_y + 1; m++)
352  {
353  for(int j = search_area_start_x; j < search_area_end_x + 1; j++)
354  {
355  if( abs( gradient* j + c1- m) < max_deviation_from_line) //Point is on left line
356  {
357  //Only include points that are within the image pixel limit
358  if(j > 0)
359  {
360  temp_checkpoint_startx.push_back(j);
361  }
362  else
363  {
364  temp_checkpoint_startx.push_back(0);
365  }
366 
367  if(m > 0)
368  {
369  temp_checkpoint_starty.push_back(m);
370  }
371  else
372  {
373  temp_checkpoint_starty.push_back(0);
374  }
375  }
376  else if ( abs( gradient* j + c2 - m) < max_deviation_from_line)//Point is on the right line
377  {
378 
379  //Only include points that are within the image pixel limit
380  if(j < msg.img_width)
381  {
382  temp_checkpoint_endx.push_back(j);
383  }
384  else
385  {
386  temp_checkpoint_endx.push_back(msg.img_width);
387  }
388 
389  if(m < msg.img_height)
390  {
391  temp_checkpoint_endy.push_back(m);
392  }
393  else
394  {
395  temp_checkpoint_endy.push_back(msg.img_height);
396  }
397  }
398  }
399  }
400  }
401 
402  //Add all required elements for this particular object
403  checkpoint_startx.push_back(temp_checkpoint_startx);
404  checkpoint_starty.push_back(temp_checkpoint_starty);
405  checkpoint_endx.push_back(temp_checkpoint_endx);
406  checkpoint_endy.push_back(temp_checkpoint_endy);
407  c1_vector.push_back(c1);
408  c2_vector.push_back(c2);
409  gradient_vector.push_back(gradient);
410  }
411  }

416  int get_corner_gdi(std::vector<int> centre, float radius, cv::Mat depth_img, float centre_depth)
417  {
418  int gdi = 0;
419 
420  std::vector<std::vector<int>> centre_corners = get_border_corners(centre, radius); // Returns the top left and bottom right corners of a region with centre and border radius
421  for(int i = centre_corners[0][1]; i < centre_corners[1][1]; i ++)
422  {
423  for(int j = centre_corners[0][0]; j < centre_corners[1][0]; j ++)
424  {
425  std::vector<int> point{j,i};
426  float point_depth = depth_img.at<ushort>(point[1],point[0]); //Get depth at current pixel
427  if(in_circle(radius,centre,point))
428  {
429  /*
430  A grasp is defined as good if the height of the centre of grasp is higher than the sides of grasps. thus for every point at the side that is
431  lower than the centre, the gdi increases.
432  */
433  min_height_diff_to_grip = table_height - centre_depth ;
434  if(point_depth - centre_depth > min_height_diff_to_grip && point_depth != 0)
435  {
436  gdi++;
437  }
438  else //Zero tolerance for any pixel in finger corners that would collide with centre. can be changed in the future in order to relax the conditions
439  {
440  return 0;
441  }
442  }
443  }
444  }
445  return gdi;
446  }

451  int get_gdi_value(cv::Mat depth_img, std::vector<int> target_centre, std::vector<int> target_corner1, std::vector<int> target_corner2, std::vector<int> target_corner3, std::vector<int> target_corner4) // int centre[2], int corner1[2], int corner2[2], int corner3[2], int corner4[2]
452  {
453  int gdi = 0;
454  float radius = 0.5 * gripper_thickness;
455  //31/8: Rather than using a single centre POINT, we will use an average of the points of the centre REGION
456  float centre_depth = 0;
457  // Get average depth value of the center region of the grasp
458  float total_centre_depth = 0;
459  std::vector<std::vector<int>> centre_search_area = get_border_corners(target_centre, radius);
460  float num_points= 0;
461  for(int centre_y = centre_search_area[0][1]; centre_y < centre_search_area[1][1]; centre_y++)
462  {
463  for(int centre_x = centre_search_area[0][0]; centre_x < centre_search_area[1][0]; centre_x++)
464  {
465  std::vector<int> point{centre_x,centre_y};
466  float point_depth = depth_img.at<ushort>(point[1],point[0]);
467  if(point_depth < table_height && point_depth > 1) // Point is on object
468  {
469  total_centre_depth += point_depth;
470  num_points++;
471  }
472  }
473  }
474 
475  //If no points in the centre region is on the object. , it is not a legitimate grasp. will be changed in the future to account for weird shapes (like a torus)
476  if(num_points == 0)
477  {
478  return 0;
479  }
480  else
481  {
482  centre_depth = total_centre_depth/num_points; //Getting average depth value of the centre of the object
483  }
484 
485  if(centre_depth < table_height && centre_depth > 1) // Center of grasp is on the surface of the object.
486  {
487  /*
488  We want to sample the points at the two ends of the grasp rectangle, along the longer axis of the rectangle. Thus we need to check if 1-4 or 1-2 is
489  the short side of the rectangle.
490  */
491 
492  //Define the centre of each of the two lines representing the short edge of a grasp rectangle.
493  std::vector<int> centre1;
494  std::vector<int> centre2;
495 
496  if(abs(length(target_corner1[0],target_corner1[1], target_corner2[0], target_corner2[1]) - gripper_thickness) > abs(length(target_corner1[0],target_corner1[1], target_corner4[0],target_corner4[1]) - gripper_thickness)) // side 1-4 is the short side of the rectangle
497  {
498  // Get the midpoints of the short edges
499  centre1.push_back(int((target_corner1[0] + target_corner4[0])/2));
500  centre1.push_back(int((target_corner1[1] + target_corner4[1])/2));
501  centre2.push_back(int((target_corner3[0] + target_corner2[0])/2));
502  centre2.push_back(int((target_corner3[1] + target_corner2[1])/2));
503  }
504  else //side 1-2 is the short side
505  {
506  // Get the midpoints of the short edges
507  centre1.push_back(int((target_corner1[0] + target_corner2[0])/2));
508  centre1.push_back(int((target_corner1[1] + target_corner2[1])/2));
509  centre2.push_back(int((target_corner3[0] + target_corner4[0])/2));
510  centre2.push_back(int((target_corner3[1] + target_corner4[1])/2));
511  }
512  // //Iterate through the search area to find points within the grasp rectangle, nearer to the two ends of the rectangle.
513 
514  int centre1_gdi = get_corner_gdi(centre1, radius, depth_img, centre_depth);
515  int centre2_gdi = get_corner_gdi(centre2, radius, depth_img, centre_depth);
516  if(centre1_gdi > 0 && centre2_gdi > 0)// Make sure both fingers are collision free
517  {
518  gdi = centre1_gdi +centre2_gdi;
519  }
520  }
521  return gdi;
522  }

528  vector<vector<int>> find_2_points(std::vector<float> potential_corner1, std::vector<float>potential_corner2,int item_num) // Given 2 corner points of a gripper box,find the other 2 corners
529  {
530 
531  // Box representation of a gripper
532  // <-------gripper width------->
533  // corner_1 _____________________________
534  // | | ^
535  // | | |
536  // | | gripper thickness
537  // | | |
538  // |_____________________________| v
539  // corner_2
540  vector<vector<int>> points;
541  std::vector<std::vector<int>> points_1;
542  std::vector<std::vector<int>> points_2;
543  /*
544  For a circle with center at one known corner, and radius gripper_thickness, find the intersection between the circle and each line
545  Each line intersects at 2 points, which are 2 candidates for one of the new corner point in a 2F gripper.
546  To find out which is the right point, the distance between that point and the second known corner
547  should be equal to the gripper width. THe point closer to this dimension is the next corner
548 
549  Given there are 2 lines, each line containing 1 known point, and we need to find 2 new points, each on each line, this
550  process is done twice.
551  */
552  if(abs(gradient_vector[item_num]) < min_zero_angle) // When the object makes an angle of 0 radians wrt x axis, gradient is ~ 0
553  {
554  /*
555  For a object with 0 rad wrt x axis, the 2 lines are horizontal. the points intersected by the circle will be the x offset of magnitude <gripper thickness>
556  y coordinate of the new point is constant since it is horizontal.
557  */
558 
559  points_1.push_back({int(round(potential_corner1[0] - gripper_thickness)) , int(round(potential_corner1[1])) });
560  points_1.push_back({int(round(potential_corner1[0] + gripper_thickness)), int(round(potential_corner1[1])) });
561 
562  points_2.push_back({int(round(potential_corner2[0] - gripper_thickness)) , int(round(potential_corner2[1])) });
563  points_2.push_back({int(round(potential_corner2[0] + gripper_thickness)), int(round(potential_corner2[1])) });
564  }
565  else if(abs(1/gradient_vector[item_num]) < min_zero_angle) // When the object makes an angle of pi/2 radians wrt x axis, gradient is ~ infinity
566  {
567  //std::cout << "vertical" <<std::endl;
568 
569  /*
570  For a object with pi/2 rad wrt x axis, the 2 lines are vertical. the points intersected by the circle will be the y offset of magnitude <gripper thickness>
571  x coordinate of the new point is constant since it is vertical.
572  */
573 
574  points_1.push_back({int(round(potential_corner1[0])) , int(round(potential_corner1[1] - gripper_thickness)) });
575  points_1.push_back({int(round(potential_corner1[0])) , int(round(potential_corner1[1] + gripper_thickness)) });
576 
577  points_2.push_back({int(round(potential_corner2[0])) , int(round(potential_corner2[1] - gripper_thickness)) });
578  points_2.push_back({int(round(potential_corner2[0])) , int(round(potential_corner2[1] + gripper_thickness)) });
579 
580  }
581  else // For any other angle, a more complicated approach is needed.
582  {
584 
585  points_1 = circle_line_intersect(gradient_vector[item_num], c1_vector[item_num], gripper_thickness, potential_corner1);
586  points_2 = circle_line_intersect(gradient_vector[item_num], c2_vector[item_num], gripper_thickness, potential_corner2);
587  }
588 
589  if(!points_1.empty() && !points_2.empty())
590  {
591  /*
592  For point derived with gripper width, the distance between the new points in line 1 and the known point in line 2 should be equal to distance_between_fingers
593  The distance for each of the new point in line 1 will be compared with gripper width, the one with the smallest difference is the right point
594  */
595 
596  if(abs(length(potential_corner2[0],potential_corner2[1],points_1[0][0],points_1[0][1]) - distance_between_fingers) < abs(length(potential_corner2[0],potential_corner2[1],points_1[1][0],points_1[1][1]) - distance_between_fingers))
597  {
598  points.push_back(points_1[0]);
599  }
600  else
601  {
602  points.push_back(points_1[1]);
603  }
604 
605  if(abs(length(potential_corner1[0],potential_corner1[1],points_2[0][0],points_2[0][1]) - distance_between_fingers) < abs(length(potential_corner1[0],potential_corner1[1],points_2[1][0],points_2[1][1]) - distance_between_fingers))
606  {
607  points.push_back(points_2[0]);
608  }
609  else
610  {
611  points.push_back(points_2[1]);
612  }
613 
614  }
615  else
616  {
617  return points; //the line is a tangent, cannot generate any corner
618  }
619 
620  if(points[0][0] < 0 || points[0][1] < 0 || points[1][0] < 0 ||points[1][1] < 0 ) //Negative coordinates are not valid
621  {
622  points.clear();
623  return points;
624  }
625  return points;
626  }

631  int best_grasp_from_point(std::vector<int> point ,Msg msg,cv::Mat depth_values, int item_num, int curr_max_gdi)
632  {
633  int gdi = curr_max_gdi;
634 
635  std::vector<int> potential_corner1;
636  std::vector<int> potential_corner2;
637  std::vector<int> potential_corner3;
